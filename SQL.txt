-- STEP 4 — AUTO-UPDATE updated_at TRIGGERS
-- ============================================================

CREATE OR REPLACE FUNCTION update_timestamp()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = NOW();
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER update_user_balances_timestamp
    BEFORE UPDATE ON user_balances
    FOR EACH ROW EXECUTE FUNCTION update_timestamp();

CREATE TRIGGER update_profiles_timestamp
    BEFORE UPDATE ON profiles
    FOR EACH ROW EXECUTE FUNCTION update_timestamp();

CREATE TRIGGER update_withdrawals_timestamp
    BEFORE UPDATE ON withdrawals
    FOR EACH ROW EXECUTE FUNCTION update_timestamp();


-- ============================================================
-- STEP 4b — AFTER INSERT ON auth.users → auto-create profile + balance
-- When a new user signs up, the database automatically creates:
--   1. A row in user_balances (balance 0.00)
--   2. A row in profiles (with email from auth metadata)
-- This eliminates the need for the frontend to INSERT these rows.
-- ============================================================

CREATE OR REPLACE FUNCTION handle_new_user()
RETURNS TRIGGER AS $$
BEGIN
    -- Create user_balances row with zero balance
    INSERT INTO public.user_balances (user_id, balance, has_deposited)
    VALUES (NEW.id::text, 0.00, false)
    ON CONFLICT (user_id) DO NOTHING;

    -- Create profiles row with email and optional full_name from metadata
    INSERT INTO public.profiles (user_id, email, full_name)
    VALUES (
        NEW.id::text,
        COALESCE(NEW.email, ''),
        COALESCE(NEW.raw_user_meta_data ->> 'full_name', '')
    )
    ON CONFLICT (user_id) DO NOTHING;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER on_auth_user_created
    AFTER INSERT ON auth.users
    FOR EACH ROW EXECUTE FUNCTION handle_new_user();

-- ============================================================
-- STEP 7 — STRICT RLS POLICIES
--
-- RULE: Normal users can only SELECT/UPDATE their own rows (user_id = auth.uid()::text).
-- ADMIN OVERRIDE: admin@investcore.com gets full ALL access on every table.
-- For tables where users also need INSERT (reviews, review_likes, kyc, messages),
-- an additional INSERT policy scoped to their own user_id is added.
-- ============================================================

-- ── Helper: is the current JWT from the admin? ──
-- Checks MULTIPLE locations in the JWT for the admin email.
-- This is the KEY FIX: Supabase stores email in different JWT claims
-- depending on the auth provider and version. We check all of them.
-- The function is SECURITY DEFINER so it runs with elevated privileges
-- and can reliably read auth.jwt() regardless of RLS context.
-- ─────────────────────────────────────────────────

CREATE OR REPLACE FUNCTION public.is_admin()
RETURNS BOOLEAN AS $$
BEGIN
    RETURN (
        -- Primary: top-level email claim (standard Supabase JWT)
        COALESCE(auth.jwt() ->> 'email', '') = 'admin@investcore.com'
        OR
        -- Fallback 1: email inside user_metadata (some Supabase versions)
        COALESCE(auth.jwt() -> 'user_metadata' ->> 'email', '') = 'admin@investcore.com'
        OR
        -- Fallback 2: email inside app_metadata
        COALESCE(auth.jwt() -> 'app_metadata' ->> 'email', '') = 'admin@investcore.com'
    );
END;
$$ LANGUAGE plpgsql STABLE SECURITY DEFINER;

-- Grant execute on is_admin() so RLS policies can call it
GRANT EXECUTE ON FUNCTION public.is_admin() TO authenticated;
GRANT EXECUTE ON FUNCTION public.is_admin() TO anon;

-- ============================================================
-- SAFE MIGRATION BLOCK — Apply to EXISTING databases
-- ============================================================
-- This block uses IF NOT EXISTS / ADD COLUMN IF NOT EXISTS so it
-- can be run safely on an existing project without dropping tables.
-- Run this INSTEAD of the full drop-and-recreate if you want to
-- preserve existing data.
-- ============================================================

-- 1. Create withdrawals table if it doesn't exist
CREATE TABLE IF NOT EXISTS withdrawals (
    id             UUID          PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id        TEXT          NOT NULL,
    amount         DECIMAL(15,2) NOT NULL,
    method         TEXT          NOT NULL DEFAULT '',
    wallet_address TEXT          NOT NULL DEFAULT '',
    status         TEXT          NOT NULL DEFAULT 'pending',
    submitted_at   TIMESTAMPTZ   NOT NULL DEFAULT NOW(),
    updated_at     TIMESTAMPTZ   NOT NULL DEFAULT NOW()
);

-- 2. Add KYC image columns if they don't exist
ALTER TABLE kyc_submissions ADD COLUMN IF NOT EXISTS file_data_1 TEXT;
ALTER TABLE kyc_submissions ADD COLUMN IF NOT EXISTS file_data_2 TEXT;
ALTER TABLE kyc_submissions ADD COLUMN IF NOT EXISTS file_data_3 TEXT;

-- 3. Indexes for withdrawals (safe — CREATE INDEX IF NOT EXISTS)
CREATE INDEX IF NOT EXISTS idx_withdrawals_user_id ON withdrawals(user_id);
CREATE INDEX IF NOT EXISTS idx_withdrawals_status  ON withdrawals(status);

-- 4. Enable RLS on withdrawals
ALTER TABLE withdrawals ENABLE ROW LEVEL SECURITY;

-- 5. Grants for withdrawals
GRANT ALL ON withdrawals TO authenticated;

-- 6. RLS policies for withdrawals (safe — uses IF NOT EXISTS pattern)
DO $$
BEGIN
    -- Drop and recreate to ensure correct definitions
    BEGIN DROP POLICY IF EXISTS "user_select_own" ON withdrawals; EXCEPTION WHEN OTHERS THEN NULL; END;
    BEGIN DROP POLICY IF EXISTS "user_update_own" ON withdrawals; EXCEPTION WHEN OTHERS THEN NULL; END;
    BEGIN DROP POLICY IF EXISTS "user_insert_own" ON withdrawals; EXCEPTION WHEN OTHERS THEN NULL; END;
    BEGIN DROP POLICY IF EXISTS "admin_all"       ON withdrawals; EXCEPTION WHEN OTHERS THEN NULL; END;

    CREATE POLICY "user_select_own" ON withdrawals
        FOR SELECT TO authenticated
        USING (user_id = auth.uid()::text OR public.is_admin());

    CREATE POLICY "user_update_own" ON withdrawals
        FOR UPDATE TO authenticated
        USING (user_id = auth.uid()::text OR public.is_admin())
        WITH CHECK (user_id = auth.uid()::text OR public.is_admin());

    CREATE POLICY "user_insert_own" ON withdrawals
        FOR INSERT TO authenticated
        WITH CHECK (user_id = auth.uid()::text OR public.is_admin());

    CREATE POLICY "admin_all" ON withdrawals
        FOR ALL TO authenticated
        USING  (public.is_admin())
        WITH CHECK (public.is_admin());
END $$;

-- 7. Auto-update updated_at trigger for withdrawals
DO $$
BEGIN
    BEGIN DROP TRIGGER IF EXISTS update_withdrawals_timestamp ON withdrawals; EXCEPTION WHEN OTHERS THEN NULL; END;
    CREATE TRIGGER update_withdrawals_timestamp
        BEFORE UPDATE ON withdrawals
        FOR EACH ROW EXECUTE FUNCTION update_timestamp();
END $$;

-- 8. Realtime for withdrawals
ALTER TABLE withdrawals REPLICA IDENTITY FULL;
DO $$
BEGIN
    BEGIN ALTER PUBLICATION supabase_realtime ADD TABLE withdrawals; EXCEPTION WHEN OTHERS THEN NULL; END;
END $$;
